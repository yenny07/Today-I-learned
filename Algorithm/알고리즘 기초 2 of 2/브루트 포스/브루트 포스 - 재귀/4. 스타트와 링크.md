## 14889 : 스타트와 링크
### 문제
https://www.acmicpc.net/problem/14889
### 코멘트
우당탕
### 코드 
```python
from sys import stdin, maxsize
from itertools import combinations


def make_team(t):
    score = 0
    for i in t:
        for j in t:
            score += sp[i][j]
            score += sp[j][i]
    print(t)
    print(score)
    return score


N = int(stdin.readline())
sp = [list(map(int, stdin.readline().split())) for _ in range(N)]
total = 0
answer = maxsize

for i in sp:
    total += sum(i)

for t in list(combinations([n for n in range(N)], int(N/2))):
    t2 = list(set([n for n in range(N)]) - set(t))
    start = make_team(t)
    link = make_team(t2)
    answer = min(answer, start-link)


print(answer)
```
